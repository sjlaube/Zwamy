user:{
         id:Integer, (primary_key)
         name:String, (unique)
         email:String, (unique)
         first_name:String,
         last_name:String,
         images:[](imageIDs),
         view_lists:[](viewListIDs),
         friends:[](userIDs), # mutually be friends
         adds:[](userIDs), # not approved by the prospective friends
         requests:[](userIDs), # friends waiting to be accepted
         owned_clients:[] client ids,  #clients owned by the user
         playing_clients:[] client ids,  #clients the user is playing on
         playable_clients:[] client ids, #clients the user gets permitted to play on
         }

image:{
          id:Integer, (primary_key)
          name:String,
          tags:[], (Strings)
          url: String,
          token: String, 
          creator: user_id,
          datetime_created: String,
          caption:String,
          comments:[], (commentIDs)
          view_lists:[], (viewListIDs)
}

comment:{
     id:Integer, (primary_key)
     user_name:String,
     user_id:Integer,
     content:String,
     datetime:String,
}

view_list:{
          id:Integer,(primary_key)
          name:String,
          tags:[], 
          images:[order:image_id],
          creator:user_id, 
          datetime_created:String,
}

status:{
   ## for cloud sync
   id: Integer,
   user: userID,
   time_created: String,   (when the list starts to be played)
   time_updated: String,    (when last update happened)
   view_list: viewListID,
   curr_image: imageID,
}



client:{
    id: integer  #primary key
    serial_number: string (unique)
    nickname: String 
    account: String
    token: String #password for the device, auto-generated by the cloud
    owner: userID 
    playable_users: [] userIDs #managed via smartdevice or web 
    curr_user: userID   #is informed by pushing
    curr_viewlist: viewlistID   #is informed by pushing
    curr_index: the current image index in the viewlist  #is informed by pushing
   
}




index:{
   user:integer,
   image:integer,
   comment:integer,
   view_list:integer,
   client: integer,
   status:integer,
}
